   #!/bin/bash
   # .git/hooks/pre-push — block .planning/ from upstream
   # Installed by Phase 26 Plan 26-01.

   # Block staged .planning/ paths in pending commits relative to upstream:
   if git diff --cached --name-only HEAD 2>/dev/null | grep -q '^\.planning/'; then
       echo "ERROR: .planning/ files are local-only (audit trail). Unstage them."

   fi
   exit 0
