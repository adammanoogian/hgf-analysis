  """25-05 differentiability: jax.grad through parallel_scan vs sequential."""
  from __future__ import annotations
  import jax
  import jax.numpy as jnp
  import numpy as np

  from prl_hgf.fitting.parallel_scan_proto import (
      parallel_scan_likelihood,
      sequential_likelihood,
  )
  from prl_hgf.simulation.batch import simulate_batch
  from prl_hgf.config_loaders import load_pick_best_cue_config

  cfg = load_pick_best_cue_config()
  cohort = simulate_batch(cfg, n_participants=1, seed=2505)
  inputs = cohort.outcomes[0]

  # Wrap likelihoods to take a flat parameter array (jax.grad-friendly)
  def loss_par(theta_flat):
      theta = unflatten(theta_flat)
      log_lik, _ = parallel_scan_likelihood(theta, inputs)
      return -log_lik

  def loss_seq(theta_flat):
      theta = unflatten(theta_flat)
      return -sequential_likelihood(theta, inputs)

  grad_par = jax.grad(loss_par)
  grad_seq = jax.grad(loss_seq)

  # Compare at 5 random theta points
  rng = np.random.default_rng(seed=2505)
  for trial in range(5):
      theta_flat = jnp.array(rng.normal(size=(5,)))  # (omega_2, omega_3, kappa, beta, zeta)
      g_par = grad_par(theta_flat)
      g_seq = grad_seq(theta_flat)

      rel_err = float(jnp.max(jnp.abs(g_par - g_seq) / (jnp.abs(g_seq) + 1e-12)))
      print(f"Trial {trial}: theta = {theta_flat}, "
            f"|g_par - g_seq|/|g_seq| max = {rel_err:.2e}")

      assert jnp.all(jnp.isfinite(g_par)), f"Non-finite gradient: {g_par}"
      assert rel_err < 1e-3, (
          f"GRADIENT MISMATCH at trial {trial}: rel_err={rel_err:.2e} > 1e-3"
      )

  print("\nAll 5 gradient checks pass within rel-err 1e-3. OK.")
  ```

  Configurable via CLI flag --backend {proto,sequential} so the SLURM script
  can launch two parallel jobs writing to different output paths.
  """
  from __future__ import annotations
  import argparse
  import json
  import time

  import arviz as az
  import blackjax
  import jax
  import jax.numpy as jnp
  import numpy as np

  from prl_hgf.fitting.parallel_scan_proto import (
      parallel_scan_likelihood,
      sequential_likelihood,
  )
  from prl_hgf.simulation.batch import simulate_batch
  from prl_hgf.config_loaders import load_pick_best_cue_config

  parser = argparse.ArgumentParser()
  parser.add_argument("--backend", choices=["proto", "sequential"], required=True)
  parser.add_argument("--seed", type=int, default=2505)
  parser.add_argument("--n-participants", type=int, default=20)
  parser.add_argument("--num-warmup", type=int, default=1000)
  parser.add_argument("--num-samples", type=int, default=1000)
  parser.add_argument("--out", type=str, required=True)
  args = parser.parse_args()

  cfg = load_pick_best_cue_config()
  cohort = simulate_batch(cfg, n_participants=args.n_participants, seed=args.seed)

  # Build per-participant log-posterior wrapping the chosen likelihood.
  # IMPORTANT: define log_posterior_seq and log_posterior_par OUTSIDE any
  # conditional (no `if args.backend == ...` inside the body). Conditional
  # access to args inside a JIT-traced function is a JAX tracing hazard
  # (closure captures args.backend at trace time, leading to silent staleness
  # if args is mutated). Use dispatch-before-define + functools.partial.
  import functools

  def log_posterior_seq(theta_flat, inputs):
      log_prior = compute_log_prior(theta_flat)
      theta = unflatten(theta_flat)
      log_lik = sequential_likelihood(theta, inputs)
      return log_prior + log_lik

  def log_posterior_par(theta_flat, inputs):
      log_prior = compute_log_prior(theta_flat)
      theta = unflatten(theta_flat)
      log_lik, _diag = parallel_scan_likelihood(theta, inputs)
      return log_prior + log_lik

  # Dispatch BEFORE define is complete: select the function object at
  # script-top, then bind via functools.partial if needed for further vmap.
  if args.backend == "proto":
      log_posterior = log_posterior_par
  else:
      log_posterior = log_posterior_seq

  # Vmap across participants
  log_post_batched = jax.vmap(log_posterior, in_axes=(0, 0))

  # BlackJAX NUTS window_adaptation (mirrors production hierarchical.py pattern)
  rng_key = jax.random.PRNGKey(args.seed)
  init_position = ...  # initialize per participant from priors

  warmup = blackjax.window_adaptation(blackjax.nuts, log_post_batched)
  t0 = time.perf_counter()
  (state, kernel_params), _ = warmup.run(rng_key, init_position, num_steps=args.num_warmup)
  warmup_wall = time.perf_counter() - t0

  # Sample
  kernel = blackjax.nuts(log_post_batched, **kernel_params).step
  t0 = time.perf_counter()
  states = inference_loop(rng_key, kernel, state, args.num_samples)
  sample_wall = time.perf_counter() - t0

  # Save chain to .nc via ArviZ
  idata = az.from_dict(
      posterior=states_to_dict(states),
      observed_data={"outcomes": cohort.outcomes},
      sample_stats={"divergences": ..., "lp": ...}
  )
  idata.to_netcdf(args.out)

  print(f"Wrote {args.out}, warmup={warmup_wall:.1f}s, sample={sample_wall:.1f}s")
  ```
