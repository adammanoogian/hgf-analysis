  rel_errs = []

  for p in range(10):
      inputs_p = cohort.outcomes[p]  # shape (T,)
      theta_p = {k: v[p] for k, v in cohort.true_theta.items()}

  ```python
  """25-04 GPU timing benchmark: parallel-scan vs sequential at three (P, T) shapes."""
  from __future__ import annotations
  import json
  import time
  import jax
  import jax.numpy as jnp
  import numpy as np

  from prl_hgf.fitting.parallel_scan_proto import (
      parallel_scan_likelihood,
      sequential_likelihood,
  )
  from prl_hgf.simulation.batch import simulate_batch
  from prl_hgf.config_loaders import load_pick_best_cue_config

  cfg = load_pick_best_cue_config()
  shapes = [(50, 420), (300, 420), (50, 2000)]
  results = {"hardware": str(jax.devices()[0].device_kind),
             "jax_version": jax.__version__,
             "shapes": []}

  for (P, T) in shapes:
      print(f"\n=== Shape (P={P}, T={T}) ===")
      cohort = simulate_batch(cfg, n_participants=P, n_trials=T, seed=2504)

      # Vectorize across participants via vmap
      par_batch = jax.vmap(parallel_scan_likelihood, in_axes=(0, 0))
      seq_batch = jax.vmap(sequential_likelihood, in_axes=(0, 0))

      par_jit = jax.jit(par_batch)
      seq_jit = jax.jit(seq_batch)

      # Warm-up (compile)
      _ = par_jit(cohort.true_theta, cohort.outcomes)[0].block_until_ready()
      _ = seq_jit(cohort.true_theta, cohort.outcomes).block_until_ready()

      # Timed runs (>= 3 repeats; record median)
      par_walls, seq_walls = [], []
      for _ in range(3):
          t0 = time.perf_counter()
          log_lik_par, diag = par_jit(cohort.true_theta, cohort.outcomes)
          log_lik_par.block_until_ready()
          par_walls.append(time.perf_counter() - t0)

          t0 = time.perf_counter()
          log_lik_seq = seq_jit(cohort.true_theta, cohort.outcomes)
          log_lik_seq.block_until_ready()
          seq_walls.append(time.perf_counter() - t0)

      # Numerical agreement on this shape too
      rel_err = float(jnp.max(jnp.abs(log_lik_par - log_lik_seq) /
                              (jnp.abs(log_lik_seq) + 1e-12)))

      shape_result = {
          "P": P,
          "T": T,
          "par_wall_seconds": float(np.median(par_walls)),
          "seq_wall_seconds": float(np.median(seq_walls)),
          "speedup": float(np.median(seq_walls) / np.median(par_walls)),
          "K_observed": int(jnp.max(diag["K_iters"])),
          "rel_err": rel_err,
      }
      print(json.dumps(shape_result, indent=2))
      results["shapes"].append(shape_result)

  with open("results/diagnostics/hgf_pscan_proto.json", "w") as f:
      json.dump(results, f, indent=2)

  print("\nWritten to results/diagnostics/hgf_pscan_proto.json")
  ```
