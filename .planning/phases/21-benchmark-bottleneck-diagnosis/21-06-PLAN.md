  Step 3 — Compute HLO hashes:
  Write an ad-hoc Python script at
  `.planning/phases/21-benchmark-bottleneck-diagnosis/compute_hlo_hashes.py`
  (kept as a working artifact; OK to commit):
    ```python
    from __future__ import annotations

    import hashlib
    from pathlib import Path


    def main() -> None:
        """Compute sha256 hashes of HLO text-IR dumps and emit a comparison matrix.

        Scans hlo_dumps/ subdirs, picks the largest .txt per subdir (primary HLO),
        computes sha256 over the text, and prints a pairwise hash-match matrix.
        """
        phase_dir = Path(
            ".planning/phases/21-benchmark-bottleneck-diagnosis/hlo_dumps"
        )
        if not phase_dir.exists():
            print(f"ERROR: {phase_dir} does not exist")
            return

        subdirs = sorted([d for d in phase_dir.iterdir() if d.is_dir()])

        hashes: dict[str, str] = {}
        for d in subdirs:
            txt_files = sorted(d.glob("*.txt"), key=lambda p: p.stat().st_size, reverse=True)
            if not txt_files:
                hashes[d.name] = "MISSING"
                continue
            primary = txt_files[0]
            text = primary.read_bytes()
            h = hashlib.sha256(text).hexdigest()[:16]
            hashes[d.name] = h
            print(f"{d.name}: {h} ({primary.name}, {len(text)} bytes)")

        print("\n=== HLO hash comparison matrix ===\n")
        names = sorted(hashes.keys())
        # Collapse identical hashes into groups
        hash_groups: dict[str, list[str]] = {}
        for name in names:
            h = hashes[name]
            hash_groups.setdefault(h, []).append(name)
        for h, members in hash_groups.items():
            print(f"  hash {h}: {', '.join(members)}")


    if __name__ == "__main__":
        main()
    ```
