def test_json_for_html_escapes_script_close():
    """Finding #1 (research §4): ensure_ascii=True alone does NOT escape
    ``</script>``. Both defenses must be applied.

    Empirically verified in ds_env 2026-04-24: ``json.dumps({'k':
    '</script>'}, ensure_ascii=True)`` returns ``{"k": "</script>"}``
    unchanged. ``_json_for_html`` applies the ``.replace("</", "<\\/")``
    post-pass to produce the MDN-canonical safe form.

    Python source-literal convention: ``"<\\\\/script>"`` (4 backslashes
    in source) encodes the runtime string ``<\\/script>`` (1 backslash,
    1 slash). See WARNING 7 escape-depth table.
    """
    payload = {"malicious": "</script><script>alert(1)</script>"}
    out = _json_for_html(payload)

    # Positive assertion: the escape produced valid-JSON-tag-safe form.
    # Python source `"<\\/script>"` (2 backslashes in source literal) is
    # the string value `<\/script>` (1 runtime backslash, 1 slash).
    assert "<\\/script>" in out, (
        f"finding #1 defense missing: output does not contain '<\\\\/script>'. "
        f"got: {out!r}"
    )

    # Negative assertion: the literal tag-close sequence is absent.
    assert "</script>" not in out, (
        f"finding #1 defense failed: output contains literal '</script>'. "
        f"got: {out!r}"
    )

    # Round-trip: the escaped form deserializes back to the original string.
    # (The JS `JSON.parse` of `<\/script>` inside a JSON string literal
    # returns `</script>`.)
    roundtrip = json.loads(out)
    assert roundtrip["malicious"] == "</script><script>alert(1)</script>"


def test_json_for_html_escapes_unicode():
    """ensure_ascii=True still escapes non-ASCII — verify both defenses
    compose without interfering."""
    out = _json_for_html({"symbol": "ω₂"})
    # ensure_ascii escapes non-ASCII to \uXXXX; `<\/` escape leaves it alone.
    assert "\\u" in out
    assert json.loads(out)["symbol"] == "ω₂"


def test_json_for_html_handles_dataclass_asdict_output(prefit_spec):
    """Happy path: real NetworkSpec asdict output serializes cleanly."""
    out = _json_for_html(asdict(prefit_spec))
    parsed = json.loads(out)
    assert parsed["n_nodes"] == 3
    assert parsed["input_idxs"] == [0]

    """TEST-04 injection roundtrip: a marker's default body is replaced

def test_render_prefit_snapshot(snapshot, prefit_spec: NetworkSpec):
    """TEST-05 pre-fit regression: bit-exact match against committed
    snapshot. Windows line-ending normalized (research §8, §10.8).

    **Order dependency**: on a fresh clone, run once with
    ``--snapshot-update`` to GENERATE the fixture. Subsequent runs
    assert against the committed fixture.

    First-time setup:
        pytest tests/test_viz_export.py::test_render_prefit_snapshot \\
            --snapshot-update

    This will create ``tests/fixtures/viz/prefit_patrl_3level.html``.
    Review the file before committing — it is the locked reference.

    Subsequent runs fail if the rendered HTML drifts from the fixture.
    Intentional regeneration uses ``--snapshot-update``.
    """
    # On a fresh checkout where the fixture file does not exist yet,
    # pytest-snapshot's `snapshot` fixture auto-creates it on the first
    # run WITH --snapshot-update and fails otherwise. Mirror the prelude
    # from test_snapshot_fixture_sanity to short-circuit on the fresh
    # case (NIT 10 plan-checker: make order-dependency explicit).
    fixture_path = (
        Path(__file__).resolve().parent
        / "fixtures"
        / "viz"
        / "prefit_patrl_3level.html"
    )
    # Only skip when the fixture is missing AND pytest is NOT running
    # with --snapshot-update (in which case the fixture gets generated).
    is_updating = any(
        arg.startswith("--snapshot-update") for arg in [] + list(__import__("sys").argv)
    )
    if not fixture_path.exists() and not is_updating:
        pytest.skip(
            "Snapshot fixture not yet generated. Run with --snapshot-update first: "
            "pytest tests/test_viz_export.py::test_render_prefit_snapshot --snapshot-update"
        )

    snapshot.snapshot_dir = str(
        Path(__file__).resolve().parent / "fixtures" / "viz"
    )

    html = render_viewer_html(prefit_spec, offline_mode=True)
    # Windows line ending normalization — git autocrlf may convert LF
    # in the committed fixture to CRLF on checkout. Rendered string
    # from Python uses LF. Normalize both sides before compare.
    html_normalized = html.replace("\r\n", "\n")

    snapshot.assert_match(html_normalized, "prefit_patrl_3level.html")
